# Unique Districts for Paradeus (Nyto)
# Replaces standard districts for "Attractor Field" origin
district_gf_nyto_city = {
	base_buildtime = 480
	min_for_deposits_on_planet = 1
	max_for_deposits_on_planet = 1
	icon = district_gf_nyto_city
	potential = {
        uses_district_set = GF_nyto_embers_gloomy_world
        NOT = { has_planet_flag = GFnyto_Spring_wakes_up }
	}
	allow = {
        uses_district_set = GF_nyto_embers_gloomy_world
        NOT = { has_planet_flag = GFnyto_Spring_wakes_up }
		NOT = {
			is_planet_class = pc_city
		}
		NOT = {
			is_planet_class = pc_relic
		}
	}
    zone_slots = {
        slot_city_government
    }
	conversion_ratio = 1
	resources = {
		category = planet_districts_cities
		cost = {
			minerals = 500
		}
		upkeep = {
			energy = 2
		}
	}
	planet_modifier = {
		planet_housing_add = 3000
		planet_amenities_add = 500
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_housing_1
			}
		}
		modifier = {
			planet_housing_add = 1000
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = {
				has_technology = tech_housing_2
			}
		}
		modifier = {
			planet_housing_add = 1000
		}
	}
	# description = district_gf_nyto_city_desc
}
